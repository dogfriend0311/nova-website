// Save/load RTTS player to localStorage
const RTTS_SAVE_KEY = "nova_rtts_player";

function saveRTTSPlayer(player) {
  try { localStorage.setItem(RTTS_SAVE_KEY, JSON.stringify(player)); } catch(e) {}
}

function loadRTTSPlayer() {
  try { 
    const saved = localStorage.getItem(RTTS_SAVE_KEY);
    return saved ? JSON.parse(saved) : null;
  } catch(e) { return null; }
}

// Helper for mobile detection
function useIsMobile(bp = 768) {
  const [isMobile, setIsMobile] = useState(window.innerWidth < bp);
  useEffect(() => {
    const handleResize = () => setIsMobile(window.innerWidth < bp);
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, [bp]);
  return isMobile;
}

      const time = 6.0 + Math.random() * 1.5;
      score = Math.max(0, Math.min(100, Math.round((7.5 - time) * 66.7)));
      setCombineResults(prev => ({ ...prev, speed: { score, time: time.toFixed(2) } }));
    } else if (gameId === "arm") {
      // Arm strength: random velocity (85 to 105 MPH)
      const velo = 85 + Math.random() * 20;
      score = Math.max(0, Math.min(100, Math.round((velo - 85) * 5)));
      setCombineResults(prev => ({ ...prev, arm: { score, velo: Math.round(velo) } }));
    } else if (gameId === "fielding") {
      // Fielding: random number of successful plays (5 to 10 out of 10)
      const plays = 5 + Math.floor(Math.random() * 6);
      score = plays * 10;
      setCombineResults(prev => ({ ...prev, fielding: { score, plays } }));
    } else if (gameId === "iq") {
      // Baseball IQ: random score 50-100
      score = 50 + Math.floor(Math.random() * 51);
      setCombineResults(prev => ({ ...prev, iq: { score } }));
    }
    
    // Move to next combine test

      setCombineIndex(combineIndex + 1);
    } else {
      // Calculate total combine score
      const speedScore = combineResults.speed?.score || 0;
      const armScore = combineResults.arm?.score || 0;
      const fieldingScore = combineResults.fielding?.score || 0;
      const iqScore = combineResults.iq?.score || 0;
      const totalScore = (speedScore + armScore + fieldingScore + iqScore) / 4;

        combineIQ: combineResults.iq?.score || 0,
        combineScore: totalScore,
      }));
      
      setPhase("interviews");
      setInterviewIndex(0);
    }
